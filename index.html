Text Effects

  • November 18, 2016
  • Make text have different effects using if/else statements
  • Blink, Fade, Toggle
  • Below is an example of blinking
var shown = true
setInterval(function() {
  if (shown) {
    $(".blinking").css({"visibility": "hidden"});
    shown = false;
  } else {
    $(".blinking").css({"visibility": "visible"});
    shown = true;
  }
}, 800)

Copy Element

  • November 30, 2016
  • Started out with one block
  • Used .clone to copy blocks
  • Used .append to add blocks
  • Used forEach and arrays to style respective blocks
var elementsList = $("#elements-list");
var element = $(".element");

var colors = ["red", "orange", "yellow", "green", "blue", "purple", "pink"];

colors.forEach(function (color) {
  $(elementsList).append(element.clone().css("background-color", color))
});
